API Reference

Complete public API documentation for all Sentinel modules.

sentinel.env

Mock environment layer for agent testing.

MockTool

class MockTool:
    name: str
    response: Any = None
    response_fn: Callable = None
    side_effect: Exception = None
    latency_ms: float = 0.0
    error_probability: float = 0.0
    error_factory: Callable = None

    def __call__(self, **kwargs) -> Any

MockAPI

class MockAPI:
    base_url: str
    routes: Dict[str, Any]

EnvironmentBuilder

class EnvironmentBuilder:
    def mock_tool(self, name: str, **kwargs) -> "EnvironmentBuilder"
    def mock_api(self, base_url: str, **kwargs) -> "EnvironmentBuilder"
    def mock_database(self, name: str) -> "EnvironmentBuilder"
    def with_rate_limit(self, calls_per_minute: int) -> "EnvironmentBuilder"
    def build(self) -> Environment

Exceptions

class MockToolError(Exception):
    status_code: int = 500

class RateLimitError(MockToolError):
    retry_after: float = 60.0
    status_code: int = 429

class TimeoutError(MockToolError):
    status_code: int = 408

sentinel.models

Core data structures for agent traces.

class StepAction(str, Enum):
    PLAN, TOOL_CALL, REASON, RESPOND, ERROR

class ErrorSeverity(str, Enum):
    LOW, MEDIUM, HIGH, CRITICAL

class ToolCall:
    tool_name: str
    arguments: Dict[str, Any]
    result: Any = None
    duration_ms: float = 0.0
    error: Optional[str] = None

class AgentTrace:
    steps: List[Step]
    tool_calls: List[ToolCall]
    def add_step(self, step: Step) -> None
    def add_tool_call(self, tool_call: ToolCall) -> None

sentinel.assertions

20+ behavioral assertions. See Assertions for full details.

# Tool Call
assert_tool_called(trace, tool_name)
assert_tool_not_called(trace, tool_name)
assert_tool_call_order(trace, expected)
assert_tool_call_count(trace, tool_name, expected_count)
assert_no_tool_errors(trace)

# State
assert_state_consistent(trace)
assert_state_changed(trace, key)

# Resilience
assert_graceful_degradation(trace, error_count)
assert_no_silent_failure(trace, validators)

# Performance
assert_latency(trace, max_ms)
assert_step_count(trace, min_steps, max_steps)

sentinel.chaos

Failure injection for behavioral testing. See Chaos Overview.

Injectors

class ToolFailureInjector:
    def __init__(self, tool_name, failure_type="error",
                 probability=1.0, seed=None)
    def wrap(self, mock: MockTool) -> ChaosToolWrapper
    def make_validator(self) -> Callable

class NetworkPartition:
    def __init__(self, connectivity, partition_probability=1.0,
                 seed=None, heal_after_calls=None)
    def wrap(self, mock: MockTool) -> ChaosToolWrapper
    def is_reachable(self, source, target) -> bool

class ClockSkew:
    def __init__(self, skew_seconds=0.0, drift_rate=0.0,
                 affected_tools=None, seed=None)
    def get_skewed_timestamp(self, tool_name=None) -> float

class MemoryPressure:
    def __init__(self, max_context_tokens=4096,
                 pressure_threshold=0.8, eviction_strategy="fifo",
                 gc_pause_ms=0.0, oom_probability=0.0, seed=None)

class ContextDegradation:
    def __init__(self, strategy="truncation", start_step=1,
                 degradation_rate=0.1, seed=None)

class CascadingFailures:
    def __init__(self, cascade_probability=0.5,
